						FROM torneig AS T
							LEFT JOIN joc         ON _02_pk_idJocTorn = _01_pk_idJoc
						WHERE _09_datBaixaTorn IS NULL
						GROUP BY _01_pk_idTorn, _01_pk_idJoc;
